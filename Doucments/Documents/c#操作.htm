3.将字符串整数，转换为整型整数。

static void Main(string[] args)

        {

            Console.WriteLine("64位机器，请输入-999999999到999999999之间的值");

            while (Console.ReadLine() != null)

            {

                string readValue = Console.ReadLine();

                ChangeStringToInt(readValue);

            }

     }

 

private static void ChangeStringToInt(string stringValue)

        {

            int iResult = 0;

            bool isFS = false;

            for (int i = 0; i < stringValue.Length; i++)

            {

                if (i == 0 && stringValue[0] == '-')

                {

                    isFS = true;

                    continue;

                }

                int cf = GetCF(stringValue.Length - 1 - i);

                iResult += (stringValue[i] - '0') * cf;

 

            }

            if (isFS)

            {

                Console.WriteLine("转为整数的值为：" + "-" + iResult.ToString());

            }

            else

            {

                Console.WriteLine("转为整数的值为：" + iResult.ToString());

            }

        }

 

private static int GetCF(int count)

        {

            int result = 1;

            for (int i = 0; i < count; i++)

            {

                result = result * 10;

            }

            return result;

        }

static void Main(string[] args)

5.+向左移动n位，-向右移动n位

static void Main(string[] args)

        {

            Console.WriteLine("Hello World!");

            int[] sourceArray = new int[10];

            int[] resultArray = new int[10];

            string readValue = "0";

            int moveCount = 0;

            Console.WriteLine("原数组：");

            for (int i = 1; i <= 10; i++)

            {

                sourceArray[i - 1] = i;

                Console.Write(i + ";");

            }

            while (Console.ReadLine() != null)

            {

                readValue = Console.ReadLine();//移动位数，+向左移，-向右移动。

                //移位

                int.TryParse(readValue, out moveCount);

                moveCount = moveCount % sourceArray.Length;

                if (moveCount < 0)

                {

                    moveCount = sourceArray.Length + moveCount;

                }

                for (int i = 0; i < sourceArray.Length; i++)

                {

                    if (i >= moveCount)

                    {

                        resultArray[i - moveCount] = sourceArray[i];

                    }

                    else

                    {

                        resultArray[i - moveCount + sourceArray.Length] = sourceArray[i];

                    }

                }

                //输出结果

                Console.WriteLine("结果数组：");

                for (int i = 0; i < resultArray.Length; i++)

                {

                    Console.Write(resultArray[i] + ";");

                }

            }

 

        }
